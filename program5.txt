// Rail fence
#include <stdio.h>
#include <ctype.h>
#include <string.h>
int main()
{
    int i, j, rails, len, count, code[100][1000];
    char str[1000];

    printf("Enter the plain text: \n");
    gets(str);
    for (i = 0, j = 0; i < strlen(str); i++)
    {
        if (str[i] == ' ')
        {
            continue;
        }
        else
        {
            str[j] = str[i];
            j++;
        }
    }
    str[j] = '\0';
    len = strlen(str);
    printf("enter the number of rails(key):\n");
    scanf("%d", &rails);
    for (i = 0; i < rails; i++)
    {
        for (j = 0; j < len; j++)
        {
            code[i][j] = 0;
        }
    }
    count = 0;
    j = 0;
    while (j < len)
    {
        if (count % 2 == 0)
        {
            for (i = 0; i < rails; i++)
            {
                code[i][j] = (int)str[j];
                j++;
            }
        }
        else
        {
            for (i = rails - 2; i > 0; i--)
            {
                code[i][j] = (int)str[j];
                j++;
            }
        }
        count++;
    }
    for (i = 0; i < rails; i++)
    {
        for (j = 0; j < len; j++)
        {
            if (code[i][j] != 0 && isalnum(code[i][j]))
            {
                printf("%c", code[i][j]);
            }
        }
    }
    return 0;
}
